<script>


let currentPrice=0;



// Yahoo Finance取得

async function getPrice(){


let code =
document.getElementById("symbol").value;



let url =
"https://query1.finance.yahoo.com/v8/finance/chart/"
+code+".T";



try{


let res =
await fetch(url);



let data =
await res.json();



currentPrice =
data.chart.result[0]
.meta.regularMarketPrice;



document.getElementById("price")
.innerHTML =
currentPrice.toLocaleString()+"円";



}catch(e){


document.getElementById("price")
.innerHTML="取得失敗";


}


}



getPrice();




// 行追加

function addRow(){


let div =
document.createElement("div");


div.className="purchaseRow";


div.innerHTML=


'<button class="deleteBtn" onclick="removeRow(this)">×</button>'+

'購入単価'+
'<input class="buyPrice" type="number">'+

'購入株数'+
'<input class="shares" type="number">';



document.getElementById("purchaseRows")
.appendChild(div);


}



//削除

function removeRow(btn){


let rows =
document.getElementsByClassName(
"purchaseRow"
);



if(rows.length<=1){

alert("最低1件必要です");

return;

}


btn.parentNode.remove();


}





function calculate(){


let prices =
document.getElementsByClassName(
"buyPrice"
);


let shares =
document.getElementsByClassName(
"shares"
);



let totalCost=0;

let totalShares=0;



for(let i=0;i<prices.length;i++){


let p =
Number(prices[i].value);


let s =
Number(shares[i].value);



if(p>0 && s>0){


totalCost += p*s;

totalShares += s;


}

}



if(totalShares==0){

alert("購入履歴を入力してください");

return;

}



let average =
totalCost/totalShares;



let profit =
(currentPrice-average)
*totalShares;



document.getElementById("result")
.innerHTML=


'<div class="card">'+

"平均取得単価："+
Math.round(average).toLocaleString()+
"円<br>"+


"保有株数："+
totalShares.toLocaleString()+
"株<br>"+


"現在株価："+
currentPrice.toLocaleString()+
"円<br><br>"+


'<div class="'+
(profit>=0?"profit":"loss")
+'">'+

"含み益："+
Math.round(profit)
.toLocaleString()+
"円"+

"</div>"+


"</div>";

}
